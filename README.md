## Install

```bash
npm install --save react-parallax-header
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-parallax-header'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
